3   import java.io.BufferedReader;
4   import java.io.FileReader;
5   import java.io.IOException;
6   
7   /**
8    * Luokka tarjoaa metodeita tekstitiedostojen lukemiseen.
9    */
10  public class Reader {
11  
12      private BufferedReader br;
13  
14      /**
15       * Metodi lukee tekstitiedoston.
16       *
17       * @param   textfile        luettava tekstitiedosto
18       * @return                  tekstitiedoston sisältämät merkit
19       * @throws  IOException     jos tekstitiedostoa ei löydy
20       */
21      public String read(String textfile) throws IOException {
22          StringBuilder sb;
23          String line;
24          String str;
25          try {
26              br = new BufferedReader(new FileReader(textfile));
27              line = br.readLine();
28              sb = new StringBuilder();
29  
30              while (line != null) {
31                  sb.append(line);
32                  sb.append(System.lineSeparator());
33                  line = br.readLine();
34              }
35  
36              str = sb.toString();
37  
38          } finally {
39              br.close();
40  
41          }
42          return str;
43      }
44  
45  }
