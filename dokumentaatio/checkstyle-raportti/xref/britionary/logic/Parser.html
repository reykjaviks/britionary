6   /**
7    * Luokka tarjoaa metodeita JSON-tiedoston parsimiseen.
8    */
9   public class Parser {
10  
11      /**
12       * Metodi parsii JSON-tiedoston.
13       * 
14       * @param   json            Parsittava JSON-tiedosto
15       * @return                  Lista löydetyistä synonyymeista
16       * @throws  ParseException  Heittää poikkeuksen jos merkkijono on tyhjä
17       */
18      public static String parseJSON(String json) throws ParseException {
19          JSONObject response = new JSONObject(json);
20          HashSet<RegionalWord> wordList = Handler.handleResults(response);
21          String str = "";    
22          for (RegionalWord word : wordList) {
23              str += word.getWord() + "\n";
24          }
25          if (str == "") {
26              throw new ParseException("No British synonyms found.");
27          }
28          return str;
29      }
