3   /**
4    * Luokka tarjoaa metodin sanojen hakemiseen ja erottaa kutsujan 
5    * sovelluslogiikan alemmista metodeista.
6    */
7   public class Searcher {
8   
9       private final Fetcher fetcher;
10      private static Searcher instance = null;
11  
12      private Searcher() {
13          fetcher = new Fetcher();
14      }
15  
16      /**
17       * Metodi pitää huolen, että luokasta voi tehdä vain yhden ilmentymän 
18       * kerrallaan.
19       * 
20       * @return          Ilmentymä
21       */
22      public static Searcher getInstance() {
23          if (instance == null) {
24              instance = new Searcher();
25          }
26          return instance;
27      }
28      
29      /**
30       * Metodi etsii synonyymeja kutsumalla Parser, Fetcher ja Converter 
31       * -luokkien metodeita.
32       * 
33       * @param   word    Käyttäjän syöttämä hakusana
34       * @return          Lista löydetyistä synonyymeista
35       */
36      public String search(String word) {
37          return Parser.parseJSON(fetcher.fetchJSON(Converter.convert(word)));
38      }
39  }
